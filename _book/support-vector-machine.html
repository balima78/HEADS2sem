y <- Species

4.11.1.5 Tuning SVM parameters

set.seed(523)
svm.tune <- tune(svm, train.x=x, train.y=y, 
                 kernel="radial", ranges=list(cost=10^(-1:2), gamma=c(.5,1,2)))

print(svm.tune)
