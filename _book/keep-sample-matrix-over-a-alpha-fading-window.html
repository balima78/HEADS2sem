6 keep sample matrix over a alpha fading window

definitions and initialization

set.seed(1)
reset_stream(strA1)

w <- 500
eps <- 0.05
alpha <- eps^(1/w)

binsA <- rep(0, length(splits))
names(binsA) <- paste("<=",splits,sep="")

binsB <- rep(0, length(splits))
names(binsB) <- paste("<=",splits,sep="")

mx<-matrix(0,10,10)
rownames(mx)<-colnames(mx)<-names(binsA)

iteration

# iterative step
for (i in 1:itera){
  ai <- get_points(strA1)[1,1]
  bi <- get_points(strA1)[1,2]
  
  binA <- which(ai <= splits)[1]
  binsA[binA] <- binsA[binA] + 1
  
  binB <- which(bi <= splits)[1]
  binsB[binB] <- binsB[binB] + 1
  
  mx<-mx * alpha # multipying the matrix by alpha before updating it
  mx[binA,binB]<-mx[binA,binB] + 1 # updating the matix
  
    if (i %% 500 == 0){
    cat("\n ### Iteration",i, "\n")
    print(binsA)
    print(binsB)
  }
}
